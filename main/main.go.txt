package main


import (
    "fmt"
    "time"
)


func compute(value int) {
    for i := 0; i < value; i++ {
        time.Sleep(time.Second)
        fmt.Println(i)
    }
}

func main() {
    fmt.Println("Goroutine ")

    
    compute(10)
    compute(10)

   
    var input string
    fmt.Scanln(&input)

}
